from django.urls import path

urlpatterns = [
    path(

        views.user_profile,
        name='profile'

        'order_history/<order_number>',
        views.order_history,
        name='order_history'
    ),

        views.add_to_favourites,
        name='add_to_favourites'
    ),
    path(
        'delete_from_favourites/<int:product_id>',
        views.delete_from_favourites,

    ),
]
